---
title: "Quickstart: Add JWT auth to your React app"
description: "Go from zero to a working JWT authentication setup in five steps — create a store, wrap your app, sign users in, and check auth status with React Auth Kit."
---

This guide walks you through the minimal setup to get React Auth Kit working in a React or Vite app. By the end you'll have a store configured, your app wrapped in `AuthProvider`, a sign-in form, and an authenticated component that reads auth state.

<Note>
  Using Next.js or React Router? The core setup below is identical — see the [integration guides](/integration/nextjs) for framework-specific steps after completing this quickstart.
</Note>

<Steps>
  <Step title="Install the package">
    Add `react-auth-kit` to your project:

    <CodeGroup>

    ```bash npm
    npm install react-auth-kit
    ```

    ```bash yarn
    yarn add react-auth-kit
    ```

    ```bash pnpm
    pnpm add react-auth-kit
    ```

    </CodeGroup>
  </Step>

  <Step title="Create the auth store">
    Call `createAuthStore` once — outside any component — to configure your token storage. You can use cookies or `localStorage`.

    <Tabs>
      <Tab title="TypeScript">
        ```tsx store.ts
        import createAuthStore from 'react-auth-kit/createAuthStore';

        // Define your user state shape
        interface IUserData {
          name: string;
          uid: string;
        }

        const store = createAuthStore<IUserData>('cookie', {
          authName: '_auth',
          cookieDomain: window.location.hostname,
          cookieSecure: window.location.protocol === 'https:',
        });

        export default store;
        ```
      </Tab>
      <Tab title="JavaScript">
        ```jsx store.js
        import createAuthStore from 'react-auth-kit/createAuthStore';

        const store = createAuthStore('cookie', {
          authName: '_auth',
          cookieDomain: window.location.hostname,
          cookieSecure: window.location.protocol === 'https:',
        });

        export default store;
        ```
      </Tab>
    </Tabs>

    To use `localStorage` instead, pass `'localstorage'` as the first argument and omit the cookie options:

    ```ts
    const store = createAuthStore('localstorage', {
      authName: '_auth',
    });
    ```

    <Tip>
      The `authName` value is used as a prefix for all storage keys. Choose a name that won't clash with other cookies or `localStorage` entries in your app.
    </Tip>
  </Step>

  <Step title="Wrap your app with AuthProvider">
    Pass the store to `AuthProvider` at the root of your component tree — before any component that uses React Auth Kit hooks.

    <Tabs>
      <Tab title="TypeScript">
        ```tsx main.tsx
        import React from 'react';
        import ReactDOM from 'react-dom/client';
        import AuthProvider from 'react-auth-kit/AuthProvider';
        import store from './store';
        import App from './App';

        ReactDOM.createRoot(document.getElementById('root')!).render(
          <React.StrictMode>
            <AuthProvider store={store}>
              <App />
            </AuthProvider>
          </React.StrictMode>
        );
        ```
      </Tab>
      <Tab title="JavaScript">
        ```jsx main.jsx
        import React from 'react';
        import ReactDOM from 'react-dom/client';
        import AuthProvider from 'react-auth-kit/AuthProvider';
        import store from './store';
        import App from './App';

        ReactDOM.createRoot(document.getElementById('root')).render(
          <React.StrictMode>
            <AuthProvider store={store}>
              <App />
            </AuthProvider>
          </React.StrictMode>
        );
        ```
      </Tab>
    </Tabs>

    <Note>
      `AuthProvider` accepts an optional `fallbackPath` prop. When set, `useIsAuthenticated` automatically redirects unauthenticated users to that path (requires a router plugin).
    </Note>
  </Step>

  <Step title="Sign in with useSignIn">
    Use the `useSignIn` hook inside any component to authenticate a user. Pass the JWT access token returned by your backend, an optional token type prefix (defaults to `Bearer`), and any user data you want to store.

    <Tabs>
      <Tab title="TypeScript">
        ```tsx LoginForm.tsx
        import React, { useState } from 'react';
        import useSignIn from 'react-auth-kit/hooks/useSignIn';

        interface IUserData {
          name: string;
          uid: string;
        }

        function LoginForm() {
          const signIn = useSignIn<IUserData>();
          const [error, setError] = useState('');

          async function handleSubmit(e: React.FormEvent<HTMLFormElement>) {
            e.preventDefault();

            // Replace this with your real API call
            const response = await fetch('/api/login', {
              method: 'POST',
              body: new FormData(e.currentTarget),
            });
            const data = await response.json();

            const success = signIn({
              auth: {
                token: data.accessToken,
                type: 'Bearer',
              },
              userState: {
                name: data.user.name,
                uid: data.user.id,
              },
            });

            if (!success) {
              setError('Sign in failed');
            }
          }

          return (
            <form onSubmit={handleSubmit}>
              <input name="email" type="email" placeholder="Email" required />
              <input name="password" type="password" placeholder="Password" required />
              {error && <p>{error}</p>}
              <button type="submit">Sign in</button>
            </form>
          );
        }

        export default LoginForm;
        ```
      </Tab>
      <Tab title="JavaScript">
        ```jsx LoginForm.jsx
        import React, { useState } from 'react';
        import useSignIn from 'react-auth-kit/hooks/useSignIn';

        function LoginForm() {
          const signIn = useSignIn();
          const [error, setError] = useState('');

          async function handleSubmit(e) {
            e.preventDefault();

            // Replace this with your real API call
            const response = await fetch('/api/login', {
              method: 'POST',
              body: new FormData(e.currentTarget),
            });
            const data = await response.json();

            const success = signIn({
              auth: {
                token: data.accessToken,
                type: 'Bearer',
              },
              userState: {
                name: data.user.name,
                uid: data.user.id,
              },
            });

            if (!success) {
              setError('Sign in failed');
            }
          }

          return (
            <form onSubmit={handleSubmit}>
              <input name="email" type="email" placeholder="Email" required />
              <input name="password" type="password" placeholder="Password" required />
              {error && <p>{error}</p>}
              <button type="submit">Sign in</button>
            </form>
          );
        }

        export default LoginForm;
        ```
      </Tab>
    </Tabs>

    `useSignIn` returns a function. That function returns `true` on success and throws an `AuthKitError` if the store configuration does not match the sign-in parameters (for example, if you include a `refresh` token but did not configure `createRefresh` in the store).
  </Step>

  <Step title="Check auth status with useIsAuthenticated">
    Use `useIsAuthenticated` in any component to check whether the current user holds a valid, non-expired token.

    <Tabs>
      <Tab title="TypeScript">
        ```tsx Dashboard.tsx
        import useIsAuthenticated from 'react-auth-kit/hooks/useIsAuthenticated';

        function Dashboard() {
          const isAuthenticated = useIsAuthenticated();

          if (!isAuthenticated()) {
            return <p>You must be signed in to view this page.</p>;
          }

          return <h1>Welcome back!</h1>;
        }

        export default Dashboard;
        ```
      </Tab>
      <Tab title="JavaScript">
        ```jsx Dashboard.jsx
        import useIsAuthenticated from 'react-auth-kit/hooks/useIsAuthenticated';

        function Dashboard() {
          const isAuthenticated = useIsAuthenticated();

          if (!isAuthenticated()) {
            return <p>You must be signed in to view this page.</p>;
          }

          return <h1>Welcome back!</h1>;
        }

        export default Dashboard;
        ```
      </Tab>
    </Tabs>

    `useIsAuthenticated` returns a function (not a boolean directly). Call it — `isAuthenticated()` — to get the current status. If the token has expired, the hook signs the user out automatically and redirects to `fallbackPath` if one was set on `AuthProvider`.
  </Step>
</Steps>
